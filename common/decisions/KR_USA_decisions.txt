USA_expanding_the_league = {
	USA_invite_to_league = {
		icon = generic_political_discourse
		
		allowed = { original_tag = USA }
		
		available = {
			NOT = { has_country_flag = USA_recent_league_invite }
			OR = {
				has_war = no
				custom_trigger_tooltip = {
					tooltip = USA_same_wars_as_FROM
					has_war = yes
					any_enemy_country = { has_war_with = FROM }
				}
			}
			FROM = {
				is_subject = no
				is_in_faction = no
				OR = {
					has_war = no
					custom_trigger_tooltip = {
						tooltip = FROM_same_wars_as_us
						has_war = yes
						any_enemy_country = { has_war_with = USA }
					}
				}
			}
		}
		
		visible = {
			is_faction_leader = yes
		}
		
		target_trigger = {
			FROM = {
				exists = yes
				is_subject = no
				is_in_faction = no
				NOT = {
					has_country_flag = USA_sent_league_invite
					has_country_flag = MEX_kicked_out_league
					has_war_with = USA
				}
				OR = {
					is_in_north_america = yes
					is_in_south_america = yes
				}
				OR = {
					any_owned_state = { is_coastal = yes }
					any_neighbor_country = { is_in_faction_with = USA }
				}
				OR = {
					has_government = social_democrat
					has_government = social_liberal
					has_government = market_liberal
					has_government = social_conservative
					has_government = authoritarian_democrat
					has_government = paternal_autocrat
				}
			}
		}
		
		days_remove = 90

		fire_only_once = no

		cost = 80
		
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision USA_invite_to_league target: [From.GetName]"
			set_country_flag = USA_recent_league_invite
			FROM = {
				set_country_flag = USA_sent_league_invite
				country_event = { id = ustates.250 }
			}
		}
		
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove USA_invite_to_league target: [From.GetName]"
			clr_country_flag = USA_recent_league_invite
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 5
				FROM = {
					any_neighbor_country = { is_in_faction_with = ROOT }
				}
			}
			modifier = {
				factor = 20
				is_neighbor_of = FROM
			}
			modifier = {
				factor = 10
				FROM = { num_divisions > 10 }
			}
		}
	}
}
