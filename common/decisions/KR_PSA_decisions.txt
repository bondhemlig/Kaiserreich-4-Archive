		icon = generic_prepare_civil_war
		allowed = {	original_tag = CAL	}
		available = {
			is_subject = no
		}
		visible = {
			has_completed_focus = CAL_rainbow_war_plans
			is_subject = no
			country_exists = FROM
		}
		target_trigger = {
			FROM = {
				exists = yes
				OR = {
					tag = USA
					tag = TEX
					tag = CSA
				}
				NOT = {
					has_war_with = CAL
					is_in_faction_with = CAL
				}
			}
		}

		fire_only_once = yes
		cost = 60
		ai_will_do = {
			factor = 100
			modifier = {
				factor = 0
				has_war = yes
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision CAL_war_plan_white1"
			set_country_flag = CAL_war_plan_white_has_begun
			every_state = {
				limit = {
					is_owned_by = FROM
					is_core_of = FROM
					NOT = {
						is_core_of = CAL
						is_claimed_by = CAL
					}
				}
				add_claim_by = FROM
			}
			if = {
				limit = { FROM = { is_in_faction = yes } }
				every_country = {
					limit = { is_in_faction_with = FROM }
					reverse_add_opinion_modifier = { target = CAL modifier = embargo }
				}
				else = {
					add_opinion_modifier = { target = FROM modifier = embargo }
				}
			}
		}
	}
