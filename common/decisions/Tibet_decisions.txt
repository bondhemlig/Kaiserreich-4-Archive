		}

				tooltip = TIB_intervene_himalayas_tt_1

		}

		cost = 110
		days_remove = 15
		days_re_enable = 30

			log = "[GetDateText]: [Root.GetName]: Decision TIB_intervene_himalayas target: [From.GetName]"
			set_country_flag = { flag = TIB_preparing_attack days = 30 value = 1 }
		}
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove TIB_intervene_himalayas target: [From.GetName]"
			if = {
				limit = { FROM = { is_subject_of = ROOT } }
				annex_country = { target = FROM }
			}
			else_if = {
				limit = { FROM = { is_valid_war_target = yes } }
				create_wargoal = { type = annex_everything target = FROM }
			}
		}
		ai_will_do = {
			factor = 20
			# Do not commit suicide
			modifier = {
				factor = 0
				OR = {
					will_not_commit_suicide = yes
					any_home_area_neighbor_country = {
						has_war_with = ROOT
					}
					has_country_flag = TIB_preparing_attack
				}
			}
		}
	}
