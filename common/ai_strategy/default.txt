	enable = {
		OR = { #Either
			AND = { #Be cananda and be a factiona leader

				is_faction_leader = yes

					tag = CAN

				}
			}
		}

					}

						}
					}

					}
				}
			}

spanish_civil_war-kingdom_support= {
	enable = {
		has_country_flag = SPR_intervention
		has_global_flag = spanish_civil_war
		has_war = no
		is_subject = no
		OR = {
			OR = { #Already Has an Event
				tag = RUS
				tag = GER
				tag = NFA
				tag = FRA
				tag = ENG
				tag = SRI
				tag = SOV
				tag = MEX
				tag = BRA
				tag = ITA
				tag = POR
				tag = AUS
				tag = IRE		
			}
			AND = {
				NOT = { #Must not be asian or in a faction where the faction leader is not in asia
					is_in_asia = yes
					any_allied_country = {
						is_faction_leader = yes
						is_in_asia = yes
					}
				}
				country_exists = SPR #Spain must exists
				OR = { #Country must either not be landlocked or border the conflict
					any_owned_state = { 
						is_coastal = yes 
					}
					any_neighbor_country = {
						OR = {
							tag = SPR
							tag = CAR
							tag = CNT
						}
					}
				}
				OR = { #Either be a major power, a faction leader, or your faction leader supports spain
					is_major = yes
					is_faction_leader = yes
					AND = {
						is_in_faction = yes
						any_allied_country = {
							is_faction_leader = yes
							has_country_flag = SPR_intervention
						}			
					}

							is_faction_leader = yes
							is_in_asia = yes

						any_owned_state = { 
							is_coastal = yes 
						}
						any_neighbor_country = {
							OR = {
								tag = SPR
