ignore_PSA_rocky_mountain = {
	enable = {
		OR = {
			tag = USA
			tag = CSA
			tag = TEX
		}
		country_exists = CAL
		NOT = { has_war_with = CAL }
		has_global_flag = PSA_rockies_ceasefire
	}
	abort = {
		OR = {
			NOT = { country_exists = CAL }
			has_war_with = CAL
			has_global_flag = { flag = PSA_rockies_ceasefire days > 365 }
		}
	}

	ai_strategy = {
		type = ignore_claim
		id = "CAL"
		value = 1
	}
	ai_strategy = {
		type = contain
		id = "CAL"
		value = 200
	}
}

#Assist Delhi in civil war
entente_assist_delhi = {
	enable = {
		is_in_faction_with = DEH
		NOT = {
			has_war_with = GER
			has_war_with = ENG
		}
		DEH = {
			OR = {
				has_war_with = BHC
				has_war_with = PRF
			}
		}
	}
	abort = {
		OR = {
			NOT = { is_in_faction_with = DEH }
			DEH = { NOT = { is_in_faction_with = CAN } }
			DEH = { NOT = { has_war_with = AFG } }
			DEH = { NOT = { has_country_flag = DEH_afghan_war } }
		}
	}
	ai_strategy = {
		type = support
		id = "DEH"
		value = 200
	}
	ai_strategy = {
		type = protect
		id = "DEH"
		value = 200
	}
	ai_strategy = {
		type = invade
		id = "PRF"
		value = 500
	}
	ai_strategy = {
		type = invade
		id = "BHC"
		value = 500
	}
}

# Canada's relations with the ACW
canada_war_with_csa = {
	enable = {
		NOT = { tag = CAN }
		is_in_faction_with = CAN
		OR = {
			has_war_with = CSA
			has_war_with = TEX
			has_war_with = USA
			has_war_with = CAL
		}
	}
	abort = {
		OR = {
			NOT = { is_in_faction_with = CAN }
			NOT = {
				has_war_with = CSA
				has_war_with = TEX
				has_war_with = USA
				has_war_with = CAL
			}
		}
	}

	ai_strategy = {
		type = protect
		id = "CAN"
		value = 200
	}
	ai_strategy = {
		type = force_defend_ally_borders
		id = "CAN"
		value = 200
	}
}

usa_canada_pact = {
	enable = {
		tag = USA
		is_subject = no
		country_exists = CAN
		not_has_socialist_government = yes
		CAN = { not_has_socialist_government = yes }
		CAN = { has_country_flag = CAN_supports_usa }
	}
	abort = {
		OR = {

			CAN = { NOT = { has_country_flag = CAN_supports_usa } }
		}

		type = befriend
		id = "CAN"
		value = 100
	}

		type = protect

	}
}

entente_ignore_aus = {
	enable = {
		NOT = { tag = CAN }
		is_in_faction_with = CAN
		has_war_with = CSA
		has_war_with = TEX
		CSA = {
			any_owned_state = {
				is_coastal = yes
				is_controlled_by = CSA
			}
		}
	}
	abort = {
		OR = {
			NOT = { is_in_faction_with = CAN }
			NOT = { has_war_with = CSA }
			NOT = { has_war_with = TEX }
			NOT = {
				CSA = {
					any_owned_state = {
						is_coastal = yes
						is_controlled_by = CSA
					}
				}
			}
		}
	}

	ai_strategy = {
		type = invade
		id = "TEX"
		value = -500
	}
}
