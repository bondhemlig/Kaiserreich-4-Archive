on_actions = {
	on_startup = {
		effect = {
			AZE = {
				give_resource_rights = { state = 990 receiver = GER }
			}
		}
	}

	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			# Remove GER resource rights in AZE
			if = {
				limit = {
					OR = {
						FROM = { tag = GER }
						FROM = { tag = AZE }
					}
				}
				GER = {
					remove_resource_rights = 990
				}
			}
			# Remove OTT resource rights in JBS
			if = {
				limit = {
					OR = {
						FROM = { tag = OTT }
						FROM = { tag = JBS }
					}
				}
				OTT = {
					remove_resource_rights = 907
					remove_resource_rights = 993
					remove_resource_rights = 658
					remove_resource_rights = 678
				}
			}
			# Ikhwan being annexed
			if = {
				limit = {
					FROM = { tag = IKH }
				}
				SAU = {
					set_country_flag = IKH_revolt_2
				}
				every_state = {
					limit = { is_core_of = IKH }
					remove_core_of = IKH
				}
			}
			# Remove Mashriq cosmetic tag
			if = {
				limit = {
					FROM = {
						tag = JOR
						has_cosmetic_tag = JOR_mashriq
					}
				}
				JOR = {
					drop_cosmetic_tag = yes
					remove_state_core = 454 #Jerusalem
					remove_state_core = 455 #Ma'an
					remove_state_core = 553 #Beirut
					remove_state_core = 554 #Syria
					remove_state_core = 677 #Aleppo
					remove_state_core = 680 #Latakia
					remove_state_core = 992 #Haifa
					remove_state_core = 995 #Homs
					remove_state_core = 754 #Medina
					remove_state_core = 679 #Hejaz
				}
			}
		}
	}

	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			if = {
				limit = {
					tag = OMA
					OR = {
						is_in_faction = yes
						is_subject = yes
					}
				}
				random_other_country = {
					limit = {
						OR = {
							AND = {
								is_faction_leader = yes
								is_in_faction_with = OMA
							}
							AND = {
								has_subject = OMA
								is_in_faction = no
							}
						}
					}
					country_event = { id = saud.153 days = 2 }
				}
			}
		}
	}

	on_new_term_election = {
		effect = {
			### Egypt ###
			if = {
				limit = { tag = EGY }
				country_event = egy.5
				country_event = egy.6
				country_event = egy.137
				country_event = egy.138
			}
		}
	}

	on_war = {
		effect = {
			# Ottomans vs. Cairo Pact
			if = {
				limit = {
					tag = OTT
					has_war_with = EGY
					NOT = { has_global_flag = OTT_cairo_pact_war }
				}
				news_event = {
					id = worldnews.50
					days = 1
				}
				set_global_flag = OTT_cairo_pact_war
			}

			# Persian events
			if = {
				limit = { tag = PER }
				country_event = {
					id = pers.35
					days = 2
				}
			}
		}
	}

	on_peace = {
		effect = {
			# Persian events
			if = {
				limit = { tag = PER }
				country_event = {
					id = pers.36
					days = 2
				}
			}
		}
	}

	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					tag = JBS
					FROM = { tag = SAU }
				}
				remove_ideas = JBS_arabian_war_of_unification
				SAU = { remove_ideas = SAU_arabian_war_of_unification }
				country_event = { id = jbs.3 days = 1 }
			}
			if = {
				limit = {
					tag = SAU
					FROM = { tag = JBS }
				}
				remove_ideas = JBS_arabian_war_of_unification
				SAU = { remove_ideas = SAU_arabian_war_of_unification }
				country_event = { id = saud.2 days = 1 }
			}
		}
	}
}
