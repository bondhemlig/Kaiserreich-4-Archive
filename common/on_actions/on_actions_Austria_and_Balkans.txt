on_actions = {
	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			# Disable Romanian civil war unit lock, transfer loser's ships
			if = {
				limit = {
					OR = {
						has_idea = ROM_civil_war
						FROM = { has_idea = ROM_civil_war }
					}
					ROOT = {
						OR = {
							tag = ROM
							tag = RRM
						}
					}
					FROM = {
						OR = {
							tag = ROM
							tag = RRM
						}
					}
				}
				ROOT = {
					remove_ideas = ROM_civil_war
					country_lock_all_division_template = no
					transfer_units_fraction = {
						target = FROM
						size = 0
						navy_ratio = 1
						keep_unit_leaders_trigger = { always = yes }
					}
				}
				FROM = {
					remove_ideas = ROM_civil_war
					country_lock_all_division_template = no
				}
			}
		}
	}
	
	#FROM is war target
	on_declare_war = {
		effect = {
			# Belgrade Pact attacks Hungary during the Ausgleich War
			if = {
				limit = {
					ROOT = {
						OR = {
							tag = ROM
							tag = RRM
							tag = SER
						}
					}
					FROM = {
						tag = HUN
						has_war_with = AUS
					}
				}
				AUS = {
					country_event = { id = ausgleich.new.702 days = 2 random = 24 }
				}
			}
		}
	}
	
	on_new_term_election = {
		effect = {
			### Albania ###
			if = {
				limit = { tag = ALB }
				country_event = albania.3
				country_event = albania.30
			}
			### Austria ###
			if = {
				limit = { tag = AUS }
				country_event = aus.elections.100
			}
			### Greece ###
			if = {
				limit = { tag = GRE }
				country_event = gre.50 #Pre-Monarchy referendum
				country_event = gre.51 #Post-Monarchist Victory
				country_event = gre.52 #Post-Republican Victory
			}
			### Serbia ###
			if = {
				limit = { tag = SER }
				country_event = serbia.6
				country_event = serbia.7
				country_event = serbia.8
				country_event = serbia.9
			}
		}
	}
	
	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					has_global_flag = romanian_civil_war
					ROOT = {
						OR = {
							tag = ROM
							tag = RRM
						}
					}
					FROM = {
						OR = {
							tag = ROM
							tag = RRM
						}
					}
				}
				ROOT = {
					country_event = { id = romania.25 hours = 6 }
					country_event = { id = romania.30 hours = 6 }
				}
				SER = {
					country_event = { id = romania.15 hours = 6 }
					country_event = { id = romania.24 days = 3 random = 96 }
				}
				AUS = {
					country_event = { id = romania.24 days = 3 random = 96 }
				}
				UKR = {
					country_event = { id = romania.24 days = 3 random = 96 }
				}
				RUS = {
					country_event = { id = romania.24 days = 3 random = 96 }
				}
			}
		}
	}
	
	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			#UKR-ROM peace deal
			if = {
				limit = {
					has_war_with = FROM
					OR = {
						ROOT = { tag = UKR }
						FROM = { tag = ROM }
					}
					OR = {
						ROOT = { tag = ROM }
						FROM = { tag = UKR }
					}
				}
				country_event = { id = romania.55 hours = 6 }
				country_event = { id = romania.56 hours = 6 }
			}
		}
	}

	on_war = {
		effect = {
			# Albania war with Serbia
			if = {
				limit = {
					tag = ALB
					has_war_with = SER
					765 = {
						is_core_of = ALB
					}
					SER = {
						owns_state = 765
					}
				}
				country_event = {
					days = 5 id = albania.38
				}
			}
		}
	}
}
