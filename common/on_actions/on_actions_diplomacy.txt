on_actions = {
	on_startup = {
		effect = {
			for_each_loop = {
				array = global.china_tags_array
				var:v = {
					#Add army inadequacy idea
					if = {
						limit = {
							OR = {
								tag = QIE
								tag = FNG
							}
						}
						add_ideas = china_inadequate_army_2
						set_variable = { army_reform_var = -0.4 }
						set_variable = { china_army_reform_cost = 100 }
					}
					else = {
						add_ideas = china_inadequate_army_1
						set_variable = { army_reform_var = -0.5 }
						set_variable = { china_army_reform_cost = 150 }
					}

					#Disable manual justification in China
					for_each_loop = {
						array = global.china_tags_array
						var:v = {
							if = {
								limit = {
									NOT = { tag = PREV }
								}
								add_ai_strategy = {
									type = ignore_claim
									id = PREV
									value = 1
								}
							}
						}
					}
					#Disable for Japan
					JAP = {
						add_ai_strategy = {
							type = ignore_claim
							id = PREV
							value = 1
						}
					}
					
					#Disable for LEC
					add_ai_strategy = {
						type = ignore_claim
						id = LEC
						value = 1
					}
					LEC = {
						add_ai_strategy = {
							type = ignore_claim
							id = PREV
							value = 1
						}
					}
				}
			}
			#Special case
			TIB = {
				add_ai_strategy = {
					type = ignore_claim
					id = "BHC"
					value = 1
				}
			}
			BHC = {
				add_ai_strategy = {
					type = ignore_claim
					id = "TIB"
					value = 1
				}
			}

			#Opinion Modifiers Setup
			every_country = { country_event = kr.political.40 }

			#Faction Opinion Modifiers
			every_country = {
				limit = {
					OR = {
						is_in_faction_with = GER
						is_subject_of = GER
					}
				}
				country_event = kr.political.50
			}
			every_country = {
				limit = {
					is_in_faction_with = CAN
				}
				country_event = kr.political.52
			}
			every_country = {
				limit = {
					is_in_faction_with = FRA
				}
				country_event = kr.political.54
			}
		}
	}
}
