on_actions = {
	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			# France capitulates
			if = {
				limit = {
					tag = FRA
					NOT = { has_global_flag = FRA_capitulated }
				}
				country_event = { id = capt.40 days = 1 }
			}

			# Move the Internationale leader
			if = {
				limit = { tag = INT }
				FRA_transfer_internationale_leader = yes
			}
		}
	}
	
	#FROM is war target
	on_declare_war = {
		effect = {
			# CoF events
			if = {
				limit = {
					ROOT = { tag = FRA }
					FROM = {
						tag = BEL
						is_subject = no
						is_in_faction = no
					}
				}
				FROM = {
					country_event = {
						id = comfrance.166
						days = 2
					}
				}
			}
			if = {
				limit = {
					ROOT = { tag = FRA }
					FROM = {
						tag = HOL
						is_subject = no
						is_in_faction = no
					}
				}
				FROM = {
					country_event = {
						id = comfrance.167
						days = 2
					}
				}
			}
		}
	}
}
