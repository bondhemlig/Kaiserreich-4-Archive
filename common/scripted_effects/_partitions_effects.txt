create_german_union_cores = {
	hidden_effect = {
		every_state = {
			limit = {
				is_germany = yes
				NOT = {
					AND = {
						OR = {
							state = 5 #East Prussia
							state = 67 #Upper Silesia
							state = 86 #Posen
							state = 776 #West Prussia
							state = 777 #Masuria
						}
						OR = {
							ROOT = { tag = POL }
							POL = { is_ally_with = ROOT }
							AND = {
								10 = {
									is_owned_and_controlled_by = ROOT
									is_claim_or_core_of_ROOT = no
								}
								NOT = {
									country_exists = POL
									ROOT = { has_country_flag = POL_CONQ }
								}
							}
						}
					}
					AND = {
						state = 188 #Gumbinnen
						OR = {
							ROOT = { tag = LIT }
							LIT = { is_ally_with = ROOT }
							AND = {
								96 = {
									is_owned_and_controlled_by = ROOT
									is_claim_or_core_of_ROOT = no
								}
								NOT = {
									country_exists = LIT
									ROOT = { has_country_flag = LIT_CONQ }
								}
							}
						}
					}
					state = 8 #Luxembourg
					state = 28 #Alsace-Lorraine
					state = 779 #Nancy
					state = 741 #Wallonian Luxembourg
					state = 782 #Northern Schleswig
				}
			}
			add_core_of = GRU
		}
	}
}
