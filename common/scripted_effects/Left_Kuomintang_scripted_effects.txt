china_reform_faction = {
	if = {
		limit = {
			tag = event_target:UPC_leader
		}
		create_faction = china_upc_faction
	}
	else_if = {
		limit = {
			tag = event_target:npa_leader
		}
		create_faction = npa_faction
	}
	else_if = {
		limit = {
			tag = event_target:yunnan_R_KMT_faction_leader
		}
		create_faction = yunnan_R_KMT_faction
	}
	else_if = {
		limit = {
			tag = CHI
			has_completed_focus = CHI_form_eastasia_faction
		}
		create_faction = "East Asian National League"
	}
	else_if = {
		limit = {
			tag = QIE
			has_country_flag = created_northern_zhili_faction
		}
		create_faction = "Northern Zhili Clique"
	}
	else_if = {
		limit = {
			tag = QIE
			has_country_flag = created_zhili_faction
		}
		create_faction = "Zhili Clique"
	}
	else_if = {
		limit = {
			tag = QIE
			has_country_flag = created_qing_faction
		}
		create_faction = "Qing Sphere"
	}
	for_each_loop = {
		array = ROOT.chinese_faction_members
		ROOT = {
			add_to_faction = v
		}
	}
}
