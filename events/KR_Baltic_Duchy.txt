country_event = {
	id = balticduchy.80
	title = balticduchy.80.t
	desc = balticduchy.80.d

	fire_only_once = yes

	trigger = {
		tag = BAT
		OR = {
		BAT = {has_country_flag = BAT_SPO_POW}
		BAT = {has_country_flag = BAT_SPO_DANGER}
		BAT = {has_country_flag = BAT_SPO_DANGER_VERY}
		}
		NOT = {BAT = {has_government = national_populist} }
	}


	mean_time_to_happen = {
		days = 140
		modifier = {
			factor = 0.5
			has_country_flag = BAT_SPO_DANGER
		}
		modifier = {
			factor = 0.25
			has_country_flag = BAT_SPO_DANGER_VERY
		}
	}

	option = {
		ai_chance = { factor = 70}
		name = balticduchy.80.a
		set_politics = {
		ruling_party = national_populist
		elections_allowed = no
		}
		set_cosmetic_tag = BAT_PRI
		set_country_flag = BAT_RUS_PRI
		if = {limit = {BAT = {has_country_flag = BAT_SPO_DANGER_VERY} }
				add_political_power = -120
		}
		if = {limit = {BAT = {has_country_flag = BAT_SPO_DANGER} }
				add_political_power = -160
		}
		if = {limit = {BAT = {has_country_flag = BAT_SPO_POW} }
				add_political_power = -200
		}
		if = {limit = {BAT = {is_in_faction_with = GER} }
				GER = { remove_from_faction = BAT}
		}
		if = {limit = {BAT = {is_subject_of = GER} }
				GER = {end_puppet = BAT}
		}
		remove_ideas_with_trait = head_of_government
		add_ideas = BAT_Nikolai_Yudenich_hog_ade
		set_party_name =  {ideology = authoritarian_democrat  name = BAT_RUS_authoritarian_democrat_party long_name = BAT_RUS_authoritarian_democrat_party_long}
		set_party_name =  {ideology = paternal_autocrat  name = BAT_RUS_paternal_autocrat_party long_name = BAT_RUS_paternal_autocrat_party_long}
		hidden_effect = {GER = {country_event {id = balticduchy.182 days = 4} } }
	}

	option = {
		ai_chance = { factor = 30}
		name = balticduchy.80.b
		if = {limit = {BAT = {has_country_flag = BAT_SPO_DANGER_VERY} }
				add_political_power = -100
			add_national_unity = -0.1
			add_popularity = { ideology = national_populist popularity = -0.1 }
		}
		if = {limit = {BAT = {has_country_flag = BAT_SPO_DANGER} }
				add_political_power = -80
			add_national_unity = -0.05
			add_popularity = { ideology = national_populist popularity = -0.15 }
		}
		if = {limit = {BAT = {has_country_flag = BAT_SPO_POW} }
				add_political_power = -60
			add_popularity = { ideology = national_populist popularity = -1.00 }
		}
	}
