country_event = {
	id = uob.7
	title = uob.7.t
	desc = uob.7.d
	picture = GFX_report_event_election_vote

	trigger = {
		tag = ENG
		has_country_flag = philip_snowden_resigns
		OR = {
			AND = {
				has_idea = ENG_tuc_economy_mosley_idea
				has_idea = ENG_tuc_internal_mosley_idea
				has_idea = ENG_tuc_foreign_mosley_idea
			}
			AND = {
				has_idea = ENG_tuc_internal_mosley_idea
				has_idea = ENG_tuc_foreign_mosley_idea
				has_idea = ENG_tuc_military_mosley_idea
			}
			AND = {
				has_idea = ENG_tuc_military_mosley_idea
				has_idea = ENG_tuc_economy_mosley_idea
				has_idea = ENG_tuc_internal_mosley_idea
			}
			AND = {
				has_idea = ENG_tuc_military_mosley_idea
				has_idea = ENG_tuc_economy_mosley_idea
				has_idea = ENG_tuc_internal_mosley_idea
				has_idea = ENG_tuc_foreign_mosley_idea
			}
		}
	}
	
	fire_only_once = yes
	
	mean_time_to_happen = { days = 7 }
	
	option = {
		name = uob.7.a
		retire_country_leader = yes
		create_country_leader = {
			name = "Oswald Mosley"
			desc = "ENG_mosley_desc"
			picture = "Portrait_England_Oswald_Mosley.tga"
			expire = "1965.1.1"
			ideology = radical_socialist_subtype
			traits = { }
		}
		set_politics = {
			ruling_party = radical_socialist
		}
		hidden_effect = {
			clr_country_flag = philip_snowden_resigns
			set_country_flag = mosley_chairman
			ENG = { 
				country_event = { id = uob.101 days = 14 }
			}
		}
	}
}
