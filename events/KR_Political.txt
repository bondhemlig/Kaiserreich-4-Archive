country_event = {
	id = kr.political.18
	immediate = {log = "[GetDateText]: [Root.GetName]: event kr.political.18"}
	title = kr.political.18.t
	desc = kr.political.18.d
	picture = GFX_report_event_generic_rally_3

	trigger = {
		has_political_power > 24
		NOT = {
			has_country_flag = unpopular_government
			has_socialist_government = yes
		}
		check_variable = { var = ROOT.party_popularity@ruling_party value = 97 compare = less_than }
		OR = {
			AND = {
				has_government = authoritarian_democrat
				has_idea_with_trait = authoritarian_democrat

			AND = {
				has_government = paternal_autocrat
