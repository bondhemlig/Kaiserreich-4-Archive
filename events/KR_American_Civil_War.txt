### End of the Civil War ###
country_event = {# CSA wins the civil war
	id = amcivwar.24
	immediate = {log = "[GetDateText]: [Root.GetName]: event amcivwar.24"}
	title = amcivwar.24.t
	picture = USA_victory_celebration

	desc = {
		text = amcivwar.24.d.noceasefire
		trigger = { NOT = { has_global_flag = PSA_rockies_ceasefire } }
	}
	desc = {
		text = amcivwar.24.d.ceasefire
		trigger = { has_global_flag = PSA_rockies_ceasefire }
	}

	fire_only_once = yes

	trigger = {
		tag = CSA
		has_global_flag = USA_civil_war_begun_flag
		NOT = {
			has_war_with = CAN
			has_war_with = MEX
			has_war_with = NEE
			has_war_with = CAL
			country_exists = USA
			country_exists = TEX
		}
		OR = {
			has_global_flag = PSA_rockies_ceasefire
			NOT = { country_exists = CAL }
		}
	}

	mean_time_to_happen = {	days = 1 }

	immediate = {
		if = {
			limit = { has_global_flag = PSA_rockies_ceasefire }
			random_other_country = {
				limit = {
					exists = yes
					OR = {
						tag = CSA
						tag = USA
						tag = TEX
					}
				}
				save_event_target_as = ACW_survivor
			}
		}
	}

	option = {
		name = amcivwar.24.a
		USA_civil_war_ends = yes
		add_ideas = USA_civil_war_aftermath_idea
		add_ideas = USA_civil_war_depression_idea
		add_ideas = USA_federalist_resistance1
		add_ideas = USA_longist_resistance1
		hidden_effect = {
			add_ideas = civilian_economy
			USA_demobilize_forces = yes

			if = {
				limit = { NOT = { has_global_flag = PSA_rockies_ceasefire } }
				news_event = { id = worldnews.42 days = 1 }
			}
		}
	}
}
