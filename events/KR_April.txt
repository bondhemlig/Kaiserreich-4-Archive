country_event = { #perpetual war
	id = april.3
	immediate = {log = "[GetDateText]: [Root.GetName]: event april.3"}
	title = april.3.t
	desc = april.3.d
	picture = GFX_report_event_glorious_vehicle

	fire_only_once = no

	trigger = {
		OR = {
			tag = REB
			tag = EMP
		}
		date > 1936.1.5
		any_neighbor_country = {
			NOT = { has_war_with = ROOT }
		}
	}

	mean_time_to_happen = {
		days = 2
	}

	option = {
		name = april.3.a
		trigger = {
			tag = REB
		}
		every_country = {
			limit = {
				is_neighbor_of = REB
				NOT = {
					has_war_with = REB
				}
			}
			REB = {
				declare_war_on = {
					target = PREV
					type = annex_everything
				}
			}
		}
		if = {
			limit = {
				country_exists = RUS
				NOT = {
					has_war_with = RUS
				}
			}
			declare_war_on = {
				target = RUS
				type = annex_everything
			}
		}
	}
