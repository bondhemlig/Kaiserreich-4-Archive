# Main German capitulation event
country_event = {
	id = capt.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event capt.1"}
	title = capt.1.t
	desc = capt.1.d
	picture = GFX_report_event_german_reichstag_bombed

	fire_only_once = yes

	trigger = {
		tag = GER
		is_subject = no
		has_war = yes
		OR = {
			surrender_progress > 0.8
			AND = {
				NOT = { controls_state = 64 } #Brandenburg
				surrender_progress > 0.75
			}
			NOT = {
				controls_state = 28 #Elsass-Lothringen
				controls_state = 51 #Ostrheinland
				controls_state = 55 #Hessen
				controls_state = 50 #Wurttemberg
				controls_state = 52 #Oberbayern
				controls_state = 59 #Hannover
			}
			AND = {
				NOT = {
					controls_state = 5 #Ostpreussen
					controls_state = 66 #Niederschlesien
					controls_state = 86 #Posen
					controls_state = 62 #Pommern
					controls_state = 64 #Brandenburg
				}
				OR = {
					NOT = {
						controls_state = 55 #Hessen
						controls_state = 51 #Ostrheinland
						controls_state = 28 #Elsass-Lothringen
					}
					NOT = {
						controls_state = 52 #Oberbayern
						controls_state = 55 #Hessen
						controls_state = 60 #Thuringen
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					MAF = {
						is_subject_of = GER
						capital_scope = { is_controlled_by = MAF }
						OR = {
							has_war = no
							AND = {
								has_capitulated = no
								surrender_progress < 0.50
							}
						}
						OR = {
							is_ai = yes
							GER = { is_ai = yes }
						}
					}
				}
				set_global_flag = GER_flees_to_africa
			}
			if = {
				limit = { NOT = { controls_state = 64 } }
				64 = { save_event_target_as = last_german_state }
				else = {
					55 = { save_event_target_as = last_german_state }
				}
			}
			every_country = {
				limit = { has_war_with = GER }
				add_stability = 0.04
			}
			random_other_country = {
				limit = { has_war_with = GER }
				news_event = { id = worldnews.145 days = 1 }
			}
		}
	}

	option = {
		name = capt.1.a #Assume control of Mittelafrika. We shall fight on from there
		trigger = {
			has_global_flag = GER_flees_to_africa
		}

		if = {
			limit = { MAF = { is_ai = no } }
			MAF = { country_event = capt.2 }
			else = {
				custom_effect_tooltip = MAF_player_becomes_GEX
				hidden_effect = { country_event = capt.21 }
			}
		}
	}
	option = {
		name = capt.1.b #There is nowhere to run. Surrender!
		trigger = {
			NOT = { has_global_flag = GER_flees_to_africa }
		}
		custom_effect_tooltip = GER_will_capitulate
		hidden_effect = { country_event = capt.22 }
	}
}

			}
		}
		if = {
			limit = { JAP = { is_in_faction_with = SIA } }
			every_owned_state = {
				limit = {

			}

			limit = {

				has_war_with = SIA
				SIA = {
					OR = {
						670 = { is_controlled_by = SIA }

# Hidden event to clean up transferred German territories
country_event = {
	id = capt.11
	immediate = {log = "[GetDateText]: [Root.GetName]: event capt.11"}

	is_triggered_only = yes
	hidden = yes

	immediate = {
		if = {
			limit = { is_faction_leader = no }
			random_country = {
				limit = {
					has_country_flag = was_at_war_with_germany
					is_faction_leader = yes
					is_in_faction_with = ROOT
				}
				save_event_target_as = leader
			}
		}
		# give out cores first
		every_country = {
			limit = {
				has_country_flag = was_at_war_with_germany
				is_in_faction_with = ROOT
				any_state = {
					has_state_flag = GER_capitulation_transfer
					is_owned_by = ROOT
					is_core_of = PREV
				}
			}
			save_event_target_as = ally
			every_state = {
				limit = {
					has_state_flag = GER_capitulation_transfer
					is_owned_by = ROOT
					is_core_of = PREV
				}
				clr_state_flag = GER_capitulation_transfer
				event_target:ally = { transfer_state = PREV }
			}
		}
		# then give out claims
		every_country = {
			limit = {
				has_country_flag = was_at_war_with_germany
				is_in_faction_with = ROOT
				any_state = {
					has_state_flag = GER_capitulation_transfer
					is_owned_by = ROOT
					is_claimed_by = PREV
				}
			}
			save_event_target_as = ally
			every_state = {
				limit = {
					has_state_flag = GER_capitulation_transfer
					is_owned_by = ROOT
					is_claimed_by = PREV
				}
				clr_state_flag = GER_capitulation_transfer
				event_target:ally = { transfer_state = PREV }
			}
		}
		# give everything else to the faction leader
		if = {
			limit = {
				is_faction_leader = no
				any_country = {
					has_country_flag = was_at_war_with_germany
					is_faction_leader = yes
					is_in_faction_with = ROOT
				}
			}
			every_owned_state = {
				limit = {
					has_state_flag = GER_capitulation_transfer
					NOT = { is_core_of = ROOT }
					NOT = { is_claimed_by = ROOT }
				}
				event_target:leader = { transfer_state = PREV }
			}
		}
	}
}

	immediate = {log = "[GetDateText]: [Root.GetName]: event capt.30"}
	
	hidden = yes
	
	trigger = {
		controls_state = 16
		OR = {
			AND = {
				16 = { is_owned_by = FRA }
				has_war_with = FRA
				FRA = { has_capitulated = yes }
			}
			owns_state = 16
		}
		NOT = {
			tag = FRA
			country_exists = FRP
			has_country_flag = FRA_CONQ
			is_joke_nation = yes
		}
		# only applies to French minors if they're a puppet/in a faction
		OR = {
			is_subject = yes
			AND = {
				is_in_faction = yes
				is_faction_leader = no
			}
			NOT = {
				tag = OCC
				tag = BRI
				tag = NRM
			}
		}
		# not National France unless France exists
		OR = {
			NOT = { tag = NFA }
			country_exists = FRA
		}
		# not Spain if it's claimed the Bourbon Throne
		OR = {
			NOT = { tag = SPR }
			NOT = { has_completed_focus = SPR_bourbon_throne }
		}
	}
	
	mean_time_to_happen = {
		days = 7
		modifier = {
			factor = 4
			country_exists = FRA
		}
	}
	
	immediate = {
		# who gets the land?
		if = {
			limit = { is_in_faction_with = NFA }
			NFA = { save_event_target_as = takes_paris }
		}
		if = {
			limit = {
				is_in_faction_with = SPR
				NOT = { is_in_faction_with = GER }
				SPR = { has_completed_focus = SPR_bourbon_throne }
			}
			SPR = { save_event_target_as = takes_paris }
		}
		else_if = {
			limit = {
				is_in_faction = yes
				is_faction_leader = no
			}
			random_country = {
				limit = {
					is_in_faction_with = ROOT
					is_faction_leader = yes
				}
				save_event_target_as = takes_paris
			}
		}
		else_if = {
			limit = {
				NOT = { tag = NFA }
				is_subject = yes
			}
			overlord = { save_event_target_as = takes_paris }
		}
		else = {
			save_event_target_as = takes_paris
		}
		
		
		# divvy up France if it exists
		if = {
			limit = { country_exists = FRA }
			every_country = {
				limit = { has_war_with = FRA }
				set_country_flag = was_at_war_with_france
			}
			take_occupied_french_states = yes
			take_unoccupied_french_states = yes
			if = {
				limit = { FRA = { is_faction_leader = yes } }
				change_third_intl_leadership = yes
			}
			event_target:takes_paris = {
				annex_country = { target = FRA }
				country_event = { id = capt.32 days = 1 } #clean up
			}
			# now clean up all transferred states
			every_country = {
				limit = {
					is_faction_leader = no
					has_country_flag = was_at_war_with_france
					any_owned_state = { has_state_flag = FRA_capitulation_transfer }
				}
				country_event = capt.31
			}
			every_country = {
				limit = {
					is_faction_leader = yes
					has_country_flag = was_at_war_with_france
					any_owned_state = { has_state_flag = FRA_capitulation_transfer }
				}
				country_event = capt.31
			}
		}
		# otherwise, make sure Paris is transferred
		else_if = {
			limit = {
				event_target:takes_paris = { NOT = { owns_state = 16 } }
			}
			event_target:takes_paris = { transfer_state = 16 }
		}
		
		# now do follow-up event
		if = {
			limit = {
				event_target:takes_paris = { tag = GER }
				21 = {
					OR = {
						is_controlled_by = NFA
						controller = { is_subject_of = NFA }
					}
				}
				NOT = {
					NFA = { has_socialist_government = yes }
					GER = { has_war_with = NFA }
					has_global_flag = france_refused_bordeaux
					GER = { has_country_flag = NFA_refused_land_exchange }
					GER = { has_country_flag = GER_refused_share_france }
				}
			}
			GER = { country_event = { id = nfa.79 hours = 3 } } #Germany offers deal to NFA
		}
		else_if = {
			limit = {
				event_target:takes_paris = { tag = GER }
				OR = {
					NFA = { has_socialist_government = yes }
					GER = { has_war_with = NFA }
					has_global_flag = france_refused_bordeaux
					GER = { has_country_flag = NFA_refused_land_exchange }
					GER = { has_country_flag = GER_refused_share_france }
					21 = {
						NOT = {
							is_controlled_by = NFA
							controller = { is_subject_of = NFA }
						}
					}
				}
				NOT = {
					GER = { has_country_flag = FRA_CONQ }
				}
			}
			GER = { country_event = { id = part.6 hours = 3 } } #Germany victory over FRA
		}
		else_if = {
			limit = {
				event_target:takes_paris = {
					NOT = { tag = NFA }
					NOT = { has_country_flag = FRA_CONQ }
				}
			}

				country_event = { id = annex.157 hours = 3 }
			}
